<script>
let CONTESTS = [];

function ddayClass(d) {
  if (d <= 3) return "urgent";
  if (d <= 14) return "soon";
  return "safe";
}

function render() {
  const q = document.getElementById("search").value.trim().toLowerCase();
  const cat = document.getElementById("cat").value;
  const list = document.getElementById("list");

  const filtered = CONTESTS.filter(c => {
    const okQ = !q || c.title.toLowerCase().includes(q);
    const okCat = !cat || (c.category || "").includes(cat);
    return okQ && okCat;
  });

  if (filtered.length === 0) {
    list.innerHTML = '<div class="empty">조건에 맞는 공모전이 없어요.</div>';
    return;
  }

  list.innerHTML = filtered.map(c => `
    <a class="card" href="${c.link}" target="_blank" rel="noopener">
      <div class="card-top">
        <span class="dday ${ddayClass(c.days_left)}">${c.dday}</span>
        <span class="cat">${c.category || ""}</span>
      </div>
      <div class="title">${escapeHtml(c.title)}</div>
    </a>
  `).join("");
}

function escapeHtml(s) {
  return s.replace(/[&<>"]/g, m => ({"&":"&amp;","<":"&lt;",">":"&gt;",'"':"&quot;"}[m]));
}

async function load() {
  try {
    const res = await fetch("./data.json?t=" + Date.now());
    const data = await res.json();
    CONTESTS = data.contests || [];
    document.getElementById("meta").textContent =
      `현재 모집중 ${data.count}건 · 마지막 갱신 ${data.updated}`;
    render();
  } catch (e) {
    document.getElementById("list").innerHTML =
      '<div class="empty">데이터를 불러오지 못했어요. 잠시 후 다시 시도해주세요.</div>';
  }
}

document.getElementById("search").addEventListener("input", render);
document.getElementById("cat").addEventListener("change", render);
load();
</script>
